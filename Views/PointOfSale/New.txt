    function handleResponsiveLayout() {
        var screenWidth = window.innerWidth || document.documentElement.clientWidth;

        $('body').removeClass('mobile-layout tablet-layout desktop-layout');
        $('.pos-main-content').removeClass('mobile-layout tablet-layout desktop-layout');

            // موبايل
            $('body').addClass('mobile-layout');
            $('.pos-main-content').addClass('mobile-layout');
            $('body').css('padding-bottom', '160px');

            // تابلت

            $('.pos-main-content').addClass('tablet-layout');
            $('body').css('padding-bottom', '120px');
        } else {
            // كمبيوتر
            $('body').addClass('desktop-layout');
            $('.pos-main-content').addClass('desktop-layout');
            $('body').css('padding-bottom', '0');
        }

    // عرض الساعة
    function showTime() {
        const date = new Date();
        let h = date.getHours();
        let m = date.getMinutes();
        let s = date.getSeconds();
        let session = "AM";

        if (h == 0) {
            h = 12;
        }
        if (h >= 12) {
            session = "PM";
        }
        if (h > 12) {
            h = h - 12;
        }
        
        m = (m < 10) ? "0" + m : m;
        s = (s < 10) ? "0" + s : s;

        const time = h + ":" + m + ":" + s + " " + session;
        $('#clock, #clock-bottom').html(time);
        setTimeout(showTime, 1000);
    }

                            showCancelButton: false,
                            confirmButtonText: '@MyERP.Properties.Resources.Ok',

    function CheckManagerPassWord() {
        var PosManagerPassword = $('#PosManagerPassword').val();
        if (PosManagerPassword != null || PosManagerPassword != " ") {
            $.get('/PointOfSale/CheckManagerPassWord?posManagerPassword=' + PosManagerPassword).done(function (r) {
                if (r.success == "managerPasswordIncorrect") {
                    Swal.fire({
                        title: '@MyERP.Properties.Resources.ThePOSManagerPasswordIsIncorrect',
                        text: " ",
                        icon: 'info',
                        showCancelButton: false,
                        confirmButtonText: '@MyERP.Properties.Resources.Ok',
                        confirmButtonColor: '#3085d6'
                    });
                    $('#DiscountValue').val(0);
                    let total = 0;
                    let totalAccessoryPrice = 0;
                    $('.price').each(function () {
                        total += $(this).val() * $(this).closest('tr').find('.quantity').val();
                        let accessoryPrice = $(this).closest('tr').find('.accessoryPrice').val();
                        let itemTotalAccessoryPrice = accessoryPrice * $(this).closest('tr').find('.quantity').val();
                        $(this).closest('tr').find('.itemTotalAccessoryPrice').val(itemTotalAccessoryPrice);
                        totalAccessoryPrice += itemTotalAccessoryPrice;
                    });
                    var _perc = ($('#DiscountValue').inputmask('unmaskedvalue') * 100 / total).toFixed(2);
                    $('#VoucherDiscountPercentage').val(_perc);
                    CalculateTotals(total, totalAccessoryPrice);
                }
                else {
                    let total = 0;
                    let totalAccessoryPrice = 0;
                    $('.price').each(function () {
                        total += $(this).val() * $(this).closest('tr').find('.quantity').val();
                        let accessoryPrice = $(this).closest('tr').find('.accessoryPrice').val();
                        let itemTotalAccessoryPrice = accessoryPrice * $(this).closest('tr').find('.quantity').val();
                        $(this).closest('tr').find('.itemTotalAccessoryPrice').val(itemTotalAccessoryPrice);
                        totalAccessoryPrice += itemTotalAccessoryPrice;
                    });
                    if (($('#DiscountValue').inputmask('unmaskedvalue') / total) >= 1) {
                        $('#DiscountValue').val(0);
                    }
                    var _perc00 = ($('#DiscountValue').inputmask('unmaskedvalue') * 100 / total).toFixed(2);
                    $('#VoucherDiscountPercentage').val(_perc00);
                    CalculateTotals(total, totalAccessoryPrice);
                }
            });
        }
        else {
            Swal.fire({
                title: '@MyERP.Properties.Resources.POSManagerPasswordMustBeEntered',
                text: " ",
                icon: 'error',
                showCancelButton: false,
                confirmButtonText: '@MyERP.Properties.Resources.Ok',
                confirmButtonColor: '#3085d6'
            });
        }
        $('#PosManagerPassword').val('');
    }

    function GetTableByHallId(IsChange) {
        $.get('/Helper/GetTableByHallId?hallId=' + $('#DepartmentHallsId').val() + '&IsChange=' + IsChange).done(function (r) {
            HallTables = r;
            if (HallTables.length) {
                HallIncreasePercentage = HallTables[0].HallIncreasePercentage;
                HallIncreaseValue = HallTables[0].HallIncreaseValue;

                if (HallIncreasePercentage > 0) {
                    HallIncreasePercentage = parseFloat(HallIncreasePercentage/100);
                }
                else if (HallIncreaseValue > 0) {
                    HallIncreaseValue = parseFloat(HallIncreaseValue);
                }
                $("#TablesDiv").empty();

                if (HallTables.length > 5) {
                    var rowCount = r.length % 5 ? Math.floor((HallTables.length / 5)) + 1 : HallTables.length / 5;
                    var next = iterator(HallTables, 5);

                    for (var j = 1; j <= rowCount; j++) {
                        var tables = next();
                        var tr = '<tr>';
                        drawTablesRow(tables);
                    }
                }
                else {
                    drawTablesRow(r)
                }
                $('#tablesModal').modal("show");
            }
            else {
                $("#TablesDiv").empty();
                $('#tablesModal').modal("show");
                Swal.fire({
                    title: '@MyERP.Properties.Resources.NoTablesInThisDepartment',
                    text: " ",
                    icon: 'error',
                    showCancelButton: false,
                    confirmButtonText: '@MyERP.Properties.Resources.Ok',
                    confirmButtonColor: '#3085d6'
                });
            }
        });
    }

    function appendCustomerOptions(refill) {
        return getCustomers(refill).then(function () {
            let custumerType = $('#CustomerType').val();
            let customerOptions = '';
            let customersfiltered = customers.filter(function (i) {
                if (custumerType == 1 || custumerType == 3)
                    return i.IsBranch != true && i.IncludeFees != true
                else if (custumerType == 2)
                    return i.IncludeFees == true;
                else if (custumerType == 4)
                    return i.IsBranch == true;
            });
            for (var i = 0; i < customersfiltered.length; i++) {
                if (custumerType == 1 || custumerType == 3)
                    customerOptions += '<option value="' + customersfiltered[i].Id + '">' + customersfiltered[i].ArName + ' - ' + customersfiltered[i].Mobile + '</option>';
                else
                    customerOptions += '<option value="' + customersfiltered[i].Id + '">' + customersfiltered[i].ArName + ' - ' + customersfiltered[i].Code + '</option>';
            }
            $('#VendorOrCustomerId').empty().append(customerOptions);
            $('#VendorOrCustomerId').trigger('change');
        });
    }

    function getItemPrice(item) {
        let _item = items.filter(function (i) { return i.Id == item.val(); })[0];
        let itemId = _item.Id;
        $('#pleasewait').fadeIn();
        $.get('/PointOfSale/GetItemPrice?PosCustomerId=' + $('#PosCustomerId').val() + '&itemId=' + itemId + '&departmentId=' + $('#DepartmentId').val()).done(function (r) {
                if (r) {
                    let isFactoryChosen = false;
                    if ($('#DepartmentId').val() == 28 && '@ViewBag.ProjectName' == "Genoise") {
                        isFactoryChosen = true;
                        let itemFactoryPrice = items.filter(function (i) { return i.Id == item.val(); })[0].FactoryPrice;
                        $(item).closest('tr').find('.price').val(itemFactoryPrice);
                        if (itemFactoryPrice == 0) {
                            $(item).closest('tr').find('.price').val(0);
                            $(item).closest('tr').find('.price').prop('required', false);
                        }
                    }
                    else {
                        var PriceWithIncrease = (r.Price);

                        if (HallIncreasePercentage > 0) {
                            PriceWithIncrease = (r.Price) + ((r.Price) * parseFloat(HallIncreasePercentage));
                        }
                        else if (HallIncreaseValue > 0) {
                            PriceWithIncrease = (r.Price) + parseFloat(HallIncreaseValue);
                        }
                        $(item).closest('tr').find('.price').val(PriceWithIncrease);
                    }
                }
                else {
                    $(item).closest('tr').find('.price').val(0);
                }
                var _price = parseFloat($(item).closest('tr').find('.price').inputmask('unmaskedvalue')) * parseInt($(item).closest('tr').find('.quantity').val());
                $(item).closest('tr').find('.total').val(_price);

                var ItemTaxPercentage = 0;
                if ($('#Id').val() > 0 && item.closest('tr').find('.ItemTaxPercentage').val() > 0) {
                    ItemTaxPercentage = item.closest('tr').find('.ItemTaxPercentage').val();
                }
                else {
                    ItemTaxPercentage = _item.ItemTaxPercentage;
                }
                ItemTaxPercentage = ItemTaxPercentage != null ? ItemTaxPercentage : 0;
                var TaxValue = (parseFloat($(item).closest('tr').find('.price').inputmask('unmaskedvalue')) * parseInt(parseFloat($(item).closest('tr').find('.quantity').val()))) * (parseFloat(ItemTaxPercentage) / 100);
                item.closest('tr').find('.ItemTaxPercentage').val(ItemTaxPercentage);
                item.closest('tr').find('.ItemTaxValue').val(TaxValue);

                $(item).closest('tr').find('.price').trigger('change');
                $('#pleasewait').fadeOut();
            })
            .fail(function () {
                $(item).closest('tr').find('.price').val(0);
                $(item).closest('tr').find('.total').val(0);
                $(item).closest('tr').find('.price').trigger('change');
                $('#pleasewait').fadeOut();
            }).done(function () {
                CalculateTotals();
            });
    }

    function CalculateTotals(total, totalAccessoryPrice) {
        if (!total || !totalAccessoryPrice) {
            total = 0;
            totalAccessoryPrice = 0;
            $('.price').each(function () {
                let itemTotal = $(this).inputmask('unmaskedvalue') * $(this).closest('tr').find('.quantity').val();
                $(this).closest('tr').find('.total').val(itemTotal);
                let accessoryPrice = $(this).closest('tr').find('.accessoryPrice').inputmask('unmaskedvalue');
                let itemTotalAccessoryPrice = accessoryPrice * $(this).closest('tr').find('.quantity').val();
                $(this).closest('tr').find('.itemTotalAccessoryPrice').val(itemTotalAccessoryPrice);
                totalAccessoryPrice += itemTotalAccessoryPrice;
                total += itemTotal;
            });
        }

        if (UsePricePolicyIncludedInItemsValueAddedTax == 'True') {
            total = total / 1.15;
        }

        $('#TotalWithoutTax').val(total)
        $('#Total').val(total);
        if (UseRefundableInsurancesForAccessories == 'True') {
            $('#RefundableInsurance').val(totalAccessoryPrice);
        }
        else {
            $('#RefundableInsurance').val('');
        }
        total += $("#DeliveryCost").inputmask('unmaskedvalue')*1;
        if ($('#DiscountValue').val()) {
            total -= $('#DiscountValue').inputmask('unmaskedvalue');
        }
        $('#NetTotal').val(total);
        let customer = customers.filter((i) => { return i.Id == $('#VendorOrCustomerId').val() })[0];
        if ( customer != null &&customer.IncludeFees == true) {
            let ServiceFees = $('#ServiceFeesPercentage').val() * total;
            $('#ServiceFees').val(ServiceFees);
            total = total * 1 + ServiceFees;
        } else
            $('#ServiceFees').val('');

        if (ApplySmokingTax == 'True' && SmokingTaxPercentage > 0) {
            var perc = (total * parseFloat(SmokingTaxPercentage / 100));
            $('#SmokingTax').val(perc)
            total = total + (total * parseFloat(SmokingTaxPercentage / 100))
        }

        var ItemTaxValue = 0;
        $('.ItemTaxValue').each(function () {
            ItemTaxValue += $(this).inputmask('unmaskedvalue') * 1;
        });

        if ($('#hasTaxes').prop('checked') == true) {
            if (UsePricePolicyIncludedInItemsValueAddedTax == 'True') {
                $('#SalesTaxes').val(total * taxPercentage);
            }
        }
        else {
            $('#SalesTaxes').val('0');
        }
        total += $('#SalesTaxes').val() * 1;

        $('#TotalAfterTaxes').val(total + totalAccessoryPrice);
        $('#TotalAfterTaxes').trigger('change');
        $('#Paid').val(total.toFixed(2))
        $('#Remaining').val(($('#Paid').inputmask('unmaskedvalue') - total).toFixed(2));
    }

    $('#hasTaxes').on('click', function () {
        if (ShowTaxPercentageAndValue == 'True') {
            $('.ItemTax').show();
        }
        else {
            $('.ItemTax').hide();
        }
        CalculateTotals();
    });

    function addEdit(n,status) {
        $('#Paid').removeAttr('required');
        let valid = true;
        if ($(".form-control:invalid").length > 0) {
            $('.form-horizontal').addClass('was-validated');
            $('html, body').animate({
                scrollTop: $(".form-control:invalid").eq(0).offset().top - 80
            }, 500);
            valid = false;
        }

        if ($('#Id').val() > 0) {
            if ($('#IsClosed').val() == 'True' || $('#IsCollected').val() == 'True') {
                Swal.fire({
                    title: '@MyERP.Properties.Resources.InvoicesThatHaveBeenClosedOrCollectedCannotBeModified',
                    text: " ",
                    icon: 'error',
                    showCancelButton: false,
                    confirmButtonText: '@MyERP.Properties.Resources.Ok',
                    confirmButtonColor: '#3085d6'
                });
                valid = false;
            }
        }
        var SalesInvoiceDetails = [];
        var ItemGroupIDs = [];
        if (valid) {
            $('#PointOfSaleDetails tr').each(function () {
                var td = $(this);
                if (td.find('.quantity').val() * 1 != 0 && td.find('.item').val() > 0) {
                    var ItemPriceId = td.find('.item-price-id').val();
                    var ItemGroupId = items.filter(function (i) { return i.ItemPriceId == ItemPriceId })[0].ItemGroupId;
                    if ($.inArray(ItemGroupId, ItemGroupIDs) == -1) {
                        ItemGroupIDs.push(ItemGroupId);
                    }
                    var salesInvoiceDetail = {
                        ItemId: td.find('.item').val(),
                        ItemGroupId: ItemGroupId,
                        ItemPriceId: td.find('.item-price-id').val(),
                        ItemUnitId: td.find('.item-unit').val(),
                        UnitEquivalent: td.find('.unit-equavalent').val(),
                        Price: td.find('.price').inputmask('unmaskedvalue') * 1,
                        WareHouseId: $('#WarehouseId').val(),
                        Qty: td.find('.quantity').val(),
                        IsActive: true,
                        IsDeleted: false,
                        ExtraTypeId: td.find('.ExtraTypeId').val(),
                        RowId: td.find('.RowId').text(),
                        AccessoryPrice: td.find('.accessoryPrice').inputmask('unmaskedvalue') * 1,
                        Notes: td.next('tr').find('.ItemNotes').text(),
                        ItemTaxPercentage: td.find('.ItemTaxPercentage').val(),
                        ItemTaxValue: td.find('.ItemTaxValue').inputmask('unmaskedvalue'),
                        IsAdded: td.find('.IsAdded').prop('checked'),
                    };
                    SalesInvoiceDetails.push(salesInvoiceDetail);
                }
            });
            valid = SalesInvoiceDetails.length > 0 ? true : false;
            valid == false ?Swal.fire({title: '@MyERP.Properties.Resources.PleaseEnterFullDetails !',text: " ",icon: 'info',showCancelButton: false,confirmButtonText: '@MyERP.Properties.Resources.Ok',confirmButtonColor: '#3085d6'}) : null;
        }
        if (valid) {
            var paymentMethods = [];
            let totalPaied = 0;
            let noneCashMoney = 0;
            var creditAmount = 0;
            $('#SalesInvoicePaymentMethods tr').each(function () {
                var td = $(this);
                totalPaied += td.find('.PaymentAmount').inputmask('unmaskedvalue') * 1;
                if (td.find('.PaymentMethodId').val() != 1) {
                    noneCashMoney += td.find('.PaymentAmount').inputmask('unmaskedvalue') * 1;
                }
                var paymentMethod = {
                    SalesInvoiceId: $('#Id').val(),
                    PaymentMethodId: td.find('.PaymentMethodId').val(),
                    CashBoxId: td.find('.PaymentAmount').inputmask('unmaskedvalue') ? td.find('.CashBoxId').val() : null,
                    Amount: td.find('.PaymentAmount').inputmask('unmaskedvalue'),
                    BankId: td.find('.PaymentAmount').inputmask('unmaskedvalue') && td.find('.BankId').val() > 0 ? td.find('.BankId').val() : null,
                    BankAccountId: td.find('.PaymentAmount').inputmask('unmaskedvalue') && td.find('.BankAccountId').val() > 0 ? td.find('.BankAccountId').val() : null
                };
                if (paymentMethod.PaymentMethodId == 2)
                    creditAmount = paymentMethod.Amount;
                paymentMethods.push(paymentMethod);
            });
            paymentMethods[0].Amount = $('#TotalAfterTaxes').inputmask('unmaskedvalue') * 1 - noneCashMoney;
            let finalTotal = $('#IsInstallmentPlan').prop('checked') ? $('#TotalAfterTotalInstallment').inputmask('unmaskedvalue') : $('#TotalAfterTaxes').inputmask('unmaskedvalue');
            if (finalTotal * 1 > totalPaied) {

                Swal.fire({
                    title: '@MyERP.Properties.Resources.TheValuePaidIsLessThanRequired',
                    text: " ",
                    icon: 'info',
                    showCancelButton: false,
                    confirmButtonText: '@MyERP.Properties.Resources.Ok',
                    confirmButtonColor: '#3085d6'
                });
                valid = false;
            }
            else if (finalTotal * 1 < noneCashMoney) {
                Swal.fire({
                    title: '@MyERP.Properties.Resources.TheValueOfTheCashlessPaymentCannotBeGreaterThanTheRequiredAmount',
                    text: " ",
                    icon: 'info',
                    showCancelButton: false,
                    confirmButtonText: '@MyERP.Properties.Resources.Ok',
                    confirmButtonColor: '#3085d6'
                });
                valid = false;
            }

            {

                    scrollTop: $(".form-control:invalid").eq(0).offset().top - 80

                showCancelButton: true,

                cancelButtonColor: '#d33',

    $(document).ready(function() {
        // تطبيق التحسينات المتجاوبة
        initResponsiveFeatures();
        
        // إعداد الساعة
        showTime();
        
        // أحداث تغيير حجم النافذة
        $(window).resize(function() {
            handleResponsiveLayout();
        });
        
        // أحداث تغيير الاتجاه للموبايل
        $(window).on('orientationchange', function() {
            setTimeout(handleResponsiveLayout, 500);
        });
    });

    // تطبيق التحسينات المتجاوبة - بسيط وآمن
    function initResponsiveFeatures() {
        handleResponsiveLayout();
        addResponsiveClasses();
    }

    // إضافة كلاسات CSS - بسيط
    function addResponsiveClasses() {
        $('.card, .card-box').addClass('shadow-soft');
        $('.btn').addClass('border-radius-lg');
        $('.form-control').addClass('border-radius-lg');
    }

    // التعامل مع التخطيط المتجاوب - بسيط جداً
    function handleResponsiveLayout() {
        var screenWidth = window.innerWidth || document.documentElement.clientWidth;
        
        // إزالة كل الكلاسات القديمة
        $('body').removeClass('mobile-layout tablet-layout desktop-layout');
        $('.pos-main-content').removeClass('mobile-layout tablet-layout desktop-layout');
        
        if (screenWidth <= 767) {
            // موبايل
            $('body').addClass('mobile-layout');
            $('.pos-main-content').addClass('mobile-layout');
            $('body').css('padding-bottom', '160px');
        } else if (screenWidth <= 1199) {
            // تابلت
            $('body').addClass('tablet-layout');
            $('.pos-main-content').addClass('tablet-layout');
            $('body').css('padding-bottom', '120px');
        } else {
            // كمبيوتر
            $('body').addClass('desktop-layout');
            $('.pos-main-content').addClass('desktop-layout');
            $('body').css('padding-bottom', '0');
        }
    }

    // عرض الساعة
    function showTime() {
        const date = new Date();
        let h = date.getHours();
        let m = date.getMinutes();
        let s = date.getSeconds();
        let session = "AM";

        if (h == 0) {
            h = 12;
        }
        if (h >= 12) {
            session = "PM";
        }
        if (h > 12) {
            h = h - 12;
        }
        
        m = (m < 10) ? "0" + m : m;
        s = (s < 10) ? "0" + s : s;

        const time = h + ":" + m + ":" + s + " " + session;
        $('#clock, #clock-bottom').html(time);
        setTimeout(showTime, 1000);
    }

    // [هنا يمكن إضافة باقي الكود الأصلي من JavaScript]

        let itemPrice = $(e).closest('tr').find('.item');
        var itemId = itemPrice.val();
        
        $('#itemDetailsBody').empty();
        if (itemId) {
            $('#itemDetailsBody').append('<div class="d-flex align-items-center"><div class="spinner-border ml-auto" role="status" aria-hidden="true"></div></div>');
            $.get('/Item/GetItemQuantityInEachWarehouse?itemId=' + itemId + '&depId=' + $('#DepartmentId').val()).done(function (r) {
                $('#itemDetailsBody').empty().append(r);
            });
        }
    }

    function AlternativeItems(e) {
        let itemPrice = $(e).closest('tr').find('.item');
        var itemId = itemPrice.val();
        $(e).closest('tr').find('.IsAltItem').prop('checked', true);
        $('#AlternativeItemsModal').modal('show');
        GetAlternativeItems(itemId);
    }
